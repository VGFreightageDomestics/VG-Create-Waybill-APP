// --- FIXED 2-STEP SHARE LOGIC --- window.executePrint = async () => { const btn = document.getElementById("mainPrintBtn"); const btnSpan = btn.querySelector("span"); // STEP 2: KUNG GENERATED NA ANG FILE, SHARE NA if (readyToShareFile) { try { // Check kung kaya ng phone mo mag-share ng PDF if (navigator.canShare && navigator.canShare({ files: [readyToShareFile] })) { await navigator.share({ files: [readyToShareFile], title: 'VG Freightage Waybill', text: 'Waybill Ready' }); } else { // Kung sobrang higpit ng browser mo at bawal ang Share alert("Hindi supported ng browser mo ang Direct Share. Dinadownload ang PDF, buksan ito at i-share sa Grozzii manually."); const link = document.createElement('a'); link.href = URL.createObjectURL(readyToShareFile); link.download = readyToShareFile.name; link.click(); } } catch (err) { console.log("Share cancelled by user."); } return; // Stop here para hindi umulit ang generation } // STEP 1: GENERATION PROCESS btn.disabled = true; const start = parseInt(document.getElementById('pmStart').value); const end = parseInt(document.getElementById('pmEnd').value); const size = document.getElementById('paperSize').value; generateStickers(currentCreatedData, start, end, size); try { const pages = document.querySelectorAll('.sticker-page'); if(pages.length === 0) return; const { jsPDF } = window.jspdf; const pdfHeight = (size === '150') ? 150 : 135; const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format:[100, pdfHeight] }); for (let i = 0; i < pages.length; i++) { btnSpan.innerText = `Processing Page ${i + 1}/${pages.length}...`; // Binigyan natin ng konting delay (100ms) para makahinga ang browser await new Promise(r => setTimeout(r, 100)); const canvas = await html2canvas(pages[i], { scale: 3, backgroundColor: "#ffffff", useCORS: true }); const imgData = canvas.toDataURL('image/png', 1.0); if (i > 0) pdf.addPage(); pdf.addImage(imgData, 'PNG', 0, 0, 100, pdfHeight); } const pdfBlob = pdf.output('blob'); readyToShareFile = new File([pdfBlob], "VG_Waybill_" + currentCreatedData.waybill + ".pdf", { type: "application/pdf" }); // SUCCESS: Change button UI (Walang auto-download na magaganap dito) btn.classList.remove('btn-dark'); btn.classList.add('btn-primary'); btn.innerHTML = ` TAP TO SHARE TO GROZZII`; btn.disabled = false; } catch (err) { console.error(err); alert("Generation Error: " + err); resetPrintButton(); } }
